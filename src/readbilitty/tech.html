@RestController
@RequestMapping("/api/users")
public class UserController {

    private final UserService userService;

    @Autowired
    public UserController(UserService userService) {
        this.userService = userService;
    }

    @GetMapping("/{id}")
    public ResponseEntity getUserById(@PathVariable Long id) {
        return userService.findById(id)
            .map(user -> ResponseEntity.ok().body(user))
            .orElse(ResponseEntity.notFound().build());
    }
}

def solution(numbers):
    # 리스트 컴프리헨션 사용
    squares = [num ** 2 for num in numbers if num > 0]

    # 람다 함수와 함께 맵 사용
    result = list(map(lambda x: x + 10, squares))

    return sorted(result, reverse=True)

SQL 쿼리 예시 
SELECT
    u.username,
    COUNT(o.id) AS total_orders,
    SUM(o.total_amount) AS total_spent
FROM
    users u
JOIN
    orders o ON u.id = o.user_id
WHERE
    o.order_date >= DATE_SUB(CURRENT_DATE, INTERVAL 30 DAY)
GROUP BY
    u.id
HAVING
    total_orders > 1
ORDER BY
    total_spent DESC
LIMIT 10;

놀유니버스(NOL UNIVERSE)는 여가를 넘어 일상의 행복을 전파하는 기업입니다. 항공·숙박·패키지 등 여행은 물론 엔터테인먼트·문화·예술과 같은 모든 일상의 경험을 제공하는 혁신 플랫폼 기업입니다.

놀유니버스는 글로벌 트래블 플랫폼 '야놀자', 국내 대표 여행ᆞ엔터테인먼트 플랫폼 '인터파크', 초개인화 여행 플랫폼 '트리플', 외국인 관광객을 위한 '트리플 코리아(TRIPLE Korea)'를 운영하고 있습니다. 이들 플랫폼은 각각 방대한 데이터와 개인화 AI(인공지능) 기술력을 바탕으로 고객들에게 새로운 경험을 제공하고 있습니다. 또 플랫폼 간 초연결과 시너지를 통해 고객 각각의 경험에 흩어진 즐거움을 하나로 연결하여 일상의 의미를 더하는 서비스를 창출하고자 합니다.

관심 이유

01

AI 기반 기술과 데이터 활용

놀유니버스는 방대한 여행 데이터와 사용자 행동 패턴을 분석하여 개인화된 추천 서비스를 제공합니다. 이러한 AI 기반 기술이 실제 비즈니스에 어떻게 적용되는지 경험하고 싶습니다.

02

성장하는 여행 플랫폼 시장

코로나19 이후 여행 시장이 빠르게 회복되며, 디지털 여행 플랫폼의 중요성이 더욱 커지고 있습니다. 이러한 성장 산업에서 경력을 쌓고 싶습니다.

03

기술 기반의 혁신 문화

놀유니버스는 기술을 중심으로 지속적인 혁신을 추구하는 기업 문화를 가지고 있습니다. 기술적 도전과 성장 기회가 많은 환경에서 일하고 싶습니다.
